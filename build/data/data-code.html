Song = { song : String }

{-| A bit like `Random.Uniform` -}
Songs
    = NoSongs
    | Songs Song (List Song)
--          ^^^^       ^^^^

Songs
  (Song "Afraid")
  [(Song "Get Back"), (Song "Ship Building")]

model = {
  songs = NoSongs
}

{-| Render songs if available! -}
renderView model =
  case model.songs of
    NoSongs ->
      songlessView

    Songs first rest ->
      (view first rest)
